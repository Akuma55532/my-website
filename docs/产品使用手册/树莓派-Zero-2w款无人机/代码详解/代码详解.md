# 系统框架

flowchart LR
  subgraph HW["外部硬件 / 数据源"]

    AR_LAUNCH["aruco_detect.launch"]

  end

  FCU --> CTRL

  AR -->|"image_raw / image_annotated / fps / tracking_info"| RV
  BR -->|"position / attitude / battery / connected"| RV

  YL_LAUNCH --> YL
  YL_LAUNCH --> BR
  AR_LAUNCH --> AR
  AR_LAUNCH --> BR
  BR_LAUNCH --> BR
